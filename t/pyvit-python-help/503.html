import threading, time, os
import pyvit.can as can
from pyvit.hw import socketcan

dev = socketcan.SocketCanDev(‘can0’)
dev.start()

frame = can.Frame (0x123, [1, 2, 3, 4, 5, 6, 7, 8])

while True:
dev.send(frame)

t = threading.Thread(args=())
